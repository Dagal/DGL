32 void CGLWindow::loop()
33 {
34  while(active)
35  {
36  startTime = SDL_GetTicks();
37  while(SDL_PollEvent(&ev))
38  {
39  switch (ev.type)
40  {
41  case SDL_QUIT:
42  active = false;
43  break;
44  case SDL_KEYDOWN:
45  animation = !animation;
46  break;
47  case SDL_MOUSEBUTTONDOWN:
48  getCurrentWorld()->getCurrentScene()->getCurrentCamera()->onMouseButton(ev.button);
49  break;
50  case SDL_VIDEORESIZE:
51  cout << "Resize event" << endl;
52  onResize(ev);
53  break;
54  default:
55  break;
56  }
57  }
58 
59  currentTime = SDL_GetTicks();
60  ellapsedTime = animation?currentTime - lastTime:0;
61  lastTime = currentTime;
62 
63  glClear(GL_COLOR_BUFFER_BIT|GL_DEPTH_BUFFER_BIT);
64  glMatrixMode(GL_MODELVIEW);
65  glLoadIdentity();
66 
67  draw(ellapsedTime);
68 
69  glFlush();

71
